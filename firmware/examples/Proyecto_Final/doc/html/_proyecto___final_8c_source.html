149static void controlDeDerrames_task (void *pvParameter) {
150
151 while (1) {
152
153 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
154
155 if (control_OnOff) {
156
157 distancia=HcSr04ReadDistanceInCentimeters();
158 if (distancia<distanciaLimite_cm){
159 LedOn(LED_1);
160 LedOn(LED_2);
161 LedOn(LED_3);
162 GPIOOn(gpio5v.pin);
163 control_estado=0;
164 control_OnOff=0;
165
166
167
168 } else if (distancia<20){
169 LedOff(LED_1);
170 LedOn(LED_2);
171 LedOn(LED_3);
172 GPIOOff(gpio5v.pin);
173
174
175 }else if(distancia<distanciaGround_cm){
176 LedOff(LED_1);
177 LedOff(LED_2);
178 LedOn(LED_3);
179 GPIOOff(gpio5v.pin);
180
181
182 }else{
183 LedsOffAll();
184 GPIOOn(gpio5v.pin); //GPIO19 en cero
185 control_estado=1;
186 control_OnOff=0;
187
188 }
189
190
191
192
193 }else if(!control_OnOff){
194 GPIOOn(gpio5v.pin); //GPIO19 en cero
195 }
196
197 }
